Cisco IOS-XR BGP

community-set Primary
  23456:200
end-set

community-set Backup
  23456:50
end-set

route-policy Comm2ActionIn
  if community matches-any Primary then
    set local-preference 200
  endif
  if community matches-any Backup then
    set local-preference 50
  endif
end-policy

router bgp 23456
  neighbor 200.1.1.1
    remote-as 64511
    address-family ipv4 unicast
      route-policy Comm2ActionIn in
    exit-address-family

community-set 1Prep
  23456:1
end-set

community-set 2Prep
  23456:2
end-set

community-set 3Prep
  23456:3
end-set

route-policy Comm2ActionOut
  if community matches-any 1Prep then
    prepend as-path 23456 1
  endif
  if community matches-any 2Prep then
    prepend as-path 23456 2
  endif
  if community matches-any 3Prep then
    prepend as-path 23456 3
  endif
end-policy

router bgp 23456
  neighbor 200.2.2.2
    remote-as 64123
    address-family ipv4 unicast
      route-policy Comm2ActionOut out
    exit-address-family

   router-id Loopback 0
   adjacency-check disable
   multi-topology / single-topology
   default-information originate
   maximum-paths [x] #default is 4
